// NEXUS Community Detection — Memory Palace Builder // Runs after email processing to cluster nodes into communities // Algorithm: Label Propagation with refinement import { serve } from "https://deno.land/std@0.177.0/http/server.ts"; import { createClient } from "https://esm.sh/@supabase/supabase-js@2"; serve(async (req: Request) => { const sb = createClient( Deno.env.get("SUPABASE_URL")!, Deno.env.get("SUPABASE_SERVICE_ROLE_KEY")! ); try { // ═══ 1. LOAD ALL NODES ═══ let allNodes: any[] = []; let offset = 0; while (true) { const { data } = await sb.from("nodes").select("id, name, category, tags, links, notes, source_emails, access_count").range(offset, offset + 999); if (!data || !data.length) break; allNodes = allNodes.concat(data); offset += data.length; if (data.length < 1000) break; } if (allNodes.length < 5) { return new Response(JSON.stringify({ status: "skip", reason: "too few nodes", count: allNodes.length }), { headers: { "Content-Type": "application/json" } }); } console.log(`[Palace] Processing ${allNodes.length} nodes`); // ═══ 2. BUILD ADJACENCY + COMPUTE LINK CONFIDENCE ═══ const nodeMap = new Map(); const adjList = new Map>(); const linkConf = new Map>(); // nodeId -> {linkedId -> confidence} allNodes.forEach(n => { nodeMap.set(String(n.id), n); adjList.set(String(n.id), new Set()); linkConf.set(String(n.id), new Map()); }); // Build adjacency and compute confidence allNodes.forEach(n => { const nId = String(n.id); const nLinks = (n.links || []).map(String); const nSources = (n.source_emails || []).map((s: any) => `${s.from||''}|${s.subject||''}`.toLowerCase()); nLinks.forEach(lid => { if (!nodeMap.has(lid)) return; adjList.get(nId)!.add(lid); adjList.get(lid)!.add(nId); // Confidence: check if two nodes share source email context const linked = nodeMap.get(lid); const linkedSources = (linked?.source_emails || []).map((s: any) => `${s.from||''}|${s.subject||''}`.toLowerCase()); let conf = "inferred"; // Default: AI auto-linked // Check for shared email sources (same sender or same subject) for (const ns of nSources) { for (const ls of linkedSources) { const [nFrom, nSubj] = ns.split("|"); const [lFrom, lSubj] = ls.split("|"); if (nFrom && lFrom && nFrom === lFrom && nFrom !== "web research" && nFrom !== "chat") { conf = "extracted"; break; } if (nSubj && lSubj && nSubj.length > 10 && nSubj === lSubj) { conf = "extracted"; break; } } if (conf === "extracted") break; } // Same category + shared tags = stronger inference if (conf === "inferred" && n.category === linked?.category) { const sharedTags = (n.tags || []).filter((t: string) => (linked?.tags || []).includes(t)); if (sharedTags.length >= 2) conf = "extracted"; // Strong co-occurrence } linkConf.get(nId)!.set(lid, conf); linkConf.get(lid)!.set(nId, conf); }); }); // ═══ 3. LABEL PROPAGATION — Community Detection ═══ // Each node starts as its own community, then adopts the most common community among neighbors const communityOf = new Map(); const nodeIds = [...nodeMap.keys()]; // Initialize: each node = own community nodeIds.forEach((id, i) => communityOf.set(id, i)); // Also give category-based initial seeding for better convergence const catSeed = new Map(); let seedId = 0; allNodes.forEach(n => { const cat = n.category || "unknown"; if (!catSeed.has(cat)) catSeed.set(cat, seedId++); }); // Seed: nodes with same category AND shared tags start in same community allNodes.forEach(n => { const nId = String(n.id); const neighbors = adjList.get(nId) || new Set(); if (neighbors.size === 0) { // Isolated nodes get category-based community communityOf.set(nId, 10000 + (catSeed.get(n.category) || 0)); } }); // Run label propagation (max 20 iterations) for (let iter = 0; iter < 20; iter++) { let changed = 0; // Shuffle node order for randomness const shuffled = [...nodeIds].sort(() => Math.random() - 0.5); for (const nId of shuffled) { const neighbors = adjList.get(nId); if (!neighbors || neighbors.size === 0) continue; // Count community votes from neighbors, weighted by link confidence const votes = new Map(); neighbors.forEach(lid => { const comm = communityOf.get(lid)!; const conf = linkConf.get(nId)?.get(lid) || "inferred"; const weight = conf === "extracted" ? 3 : conf === "inferred" ? 1 : 0.5; votes.set(comm, (votes.get(comm) || 0) + weight); }); // Pick highest-voted community let bestComm = communityOf.get(nId)!; let bestVotes = 0; votes.forEach((v, c) => { if (v > bestVotes) { bestVotes = v; bestComm = c; } }); if (bestComm !== communityOf.get(nId)) { communityOf.set(nId, bestComm); changed++; } } console.log(`[Palace] Iteration ${iter + 1}: ${changed} nodes moved`); if (changed === 0) break; // Converged } // ═══ 4. RENUMBER COMMUNITIES (0, 1, 2, ...) ═══ const uniqueComms = [...new Set(communityOf.values())]; const commRemap = new Map(); uniqueComms.forEach((c, i) => commRemap.set(c, i)); nodeIds.forEach(id => communityOf.set(id, commRemap.get(communityOf.get(id)!)!)); const numCommunities = commRemap.size; console.log(`[Palace] Found ${numCommunities} communities`); // ═══ 5. IDENTIFY ROLES — God nodes, bridges, core, peripheral ═══ const commNodes = new Map(); nodeIds.forEach(id => { const c = communityOf.get(id)!; if (!commNodes.has(c)) commNodes.set(c, []); commNodes.get(c)!.push(id); }); const roles = new Map(); const godNodes = new Map(); // communityId -> godNodeId commNodes.forEach((members, commId) => { // God node = highest degree within community let maxDeg = 0, godId = members[0]; members.forEach(id => { const neighbors = adjList.get(id) || new Set(); // Count intra-community links const intraDeg = [...neighbors].filter(n => communityOf.get(n) === commId).length; if (intraDeg > maxDeg) { maxDeg = intraDeg; godId = id; } }); godNodes.set(commId, godId); roles.set(godId, "god"); // Bridge nodes = linked to multiple communities members.forEach(id => { if (roles.get(id) === "god") return; const neighbors = adjList.get(id) || new Set(); const neighborComms = new Set([...neighbors].map(n => communityOf.get(n))); neighborComms.delete(commId); // Remove own community if (neighborComms.size > 0) { roles.set(id, "bridge"); } else { const deg = (adjList.get(id)?.size || 0); roles.set(id, deg >= 3 ? "core" : "peripheral"); } }); }); // ═══ 6. BUILD COMMUNITY METADATA ═══ const communityMeta: any[] = []; commNodes.forEach((members, commId) => { // Dominant category const catCount = new Map(); members.forEach(id => { const cat = nodeMap.get(id)?.category || "unknown"; catCount.set(cat, (catCount.get(cat) || 0) + 1); }); let domCat = "unknown", domCount = 0; catCount.forEach((c, cat) => { if (c > domCount) { domCount = c; domCat = cat; } }); // Label from god node + dominant category const godId = godNodes.get(commId)!; const godName = nodeMap.get(godId)?.name || "Unknown"; const label = `${domCat.charAt(0).toUpperCase() + domCat.slice(1)}: ${godName}`; // Bridge node IDs const bridges = members.filter(id => roles.get(id) === "bridge"); // Connected communities (via bridges) const connected = new Set(); bridges.forEach(id => { const neighbors = adjList.get(id) || new Set(); neighbors.forEach(n => { const nc = communityOf.get(n); if (nc !== undefined && nc !== commId) connected.add(nc); }); }); // Color from category const CAT_COLORS: Record = { equipment: [140, 170, 240], contractors: [100, 200, 170], vendors: [210, 175, 100], procedure: [180, 140, 210], projects: [200, 180, 80], people: [160, 210, 190], systems: [130, 160, 210], parts: [190, 170, 150], location: [140, 190, 140], }; communityMeta.push({ community_id: commId, label, god_node_id: godId, god_node_name: godName, node_count: members.length, dominant_category: domCat, bridge_node_ids: bridges, connected_communities: [...connected], color_rgb: CAT_COLORS[domCat] || [200, 195, 185], updated_at: new Date().toISOString(), }); }); // ═══ 7. WRITE BACK TO DATABASE ═══ // Update nodes in batches let updated = 0; const batchSize = 50; const nodeUpdates = nodeIds.map(id => ({ id, community_id: communityOf.get(id), community_role: roles.get(id) || "peripheral", community_label: communityMeta.find(c => c.community_id === communityOf.get(id))?.label || null, link_confidence: Object.fromEntries(linkConf.get(id) || new Map()), })); for (let i = 0; i < nodeUpdates.length; i += batchSize) { const batch = nodeUpdates.slice(i, i + batchSize); for (const upd of batch) { const { error } = await sb.from("nodes").update({ community_id: upd.community_id, community_role: upd.community_role, community_label: upd.community_label, link_confidence: upd.link_confidence, }).eq("id", upd.id); if (!error) updated++; } } // Upsert community metadata for (const cm of communityMeta) { await sb.from("communities").upsert(cm, { onConflict: "community_id" }); } // ═══ 8. GENERATE COMMUNITY SUMMARIES (GraphRAG Layer) ═══ const apiKey = Deno.env.get("ANTHROPIC_API_KEY"); let summariesGenerated = 0; if (apiKey) { // Only summarize significant communities (5+ nodes), top 20 by size const toSummarize = communityMeta .filter(c => c.node_count >= 5) .sort((a, b) => b.node_count - a.node_count) .slice(0, 20); for (const comm of toSummarize) { try { const members = commNodes.get(comm.community_id) || []; const topNodes = members .map(id => nodeMap.get(id)) .filter(Boolean) .sort((a, b) => (b.access_count || 0) - (a.access_count || 0)) .slice(0, 12); const nodeText = topNodes.map(n => `${n.name} (${n.category}): ${(n.notes || '').slice(0, 150)}` ).join('\n'); const resp = await fetch("https://api.anthropic.com/v1/messages", { method: "POST", headers: { "Content-Type": "application/json", "x-api-key": apiKey, "anthropic-version": "2023-06-01", }, body: JSON.stringify({ model: "claude-sonnet-4-20250514", max_tokens: 200, messages: [{ role: "user", content: `These are related items from a restaurant operations system. Summarize in 2-3 sentences how they connect. Focus on relationships, dependencies, and what matters operationally. Be specific with names and details. No bullet points.\n\n${nodeText}` }] }) }); const data = await resp.json(); const summary = data.content?.[0]?.text || ''; if (summary.length > 20) { await sb.from("communities").update({ summary }).eq("community_id", comm.community_id); summariesGenerated++; } } catch (e) { console.error(`[Palace] Summary failed for community ${comm.community_id}:`, e); } } console.log(`[Palace] Generated ${summariesGenerated} community summaries`); } // ═══ 9. UPDATE TEMPORAL DATA ═══ // Backfill last_relevant_date for nodes that don't have it try { await sb.rpc('', {}).catch(() => {}); // No-op, just checking connection // Update via raw SQL would be better but we'll do it node-by-node for critical ones for (const node of allNodes.filter(n => !n.last_relevant_date)) { const sources = node.source_emails || []; if (sources.length) { const dates = sources.map((s: any) => s.date).filter(Boolean).sort().reverse(); if (dates[0]) { await sb.from("nodes").update({ last_relevant_date: dates[0] }).eq("id", node.id); } } } } catch (e) { console.log("[Palace] Temporal backfill skipped:", e); } // Update last run timestamp await sb.from("nexus_config").update({ config: sb.rpc ? undefined : { last_community_run: new Date().toISOString() } }).eq("id", 2).catch(() => {}); console.log(`[Palace] Done. ${updated} nodes updated, ${communityMeta.length} communities saved, ${summariesGenerated} summaries generated`); return new Response(JSON.stringify({ status: "ok", nodes_processed: allNodes.length, nodes_updated: updated, communities: communityMeta.length, summaries_generated: summariesGenerated, community_summary: communityMeta.map(c => ({ id: c.community_id, label: c.label, nodes: c.node_count, bridges: c.bridge_node_ids.length, connections: c.connected_communities.length, })), }), { headers: { "Content-Type": "application/json" } }); } catch (err) { console.error("[Palace] Error:", err); return new Response(JSON.stringify({ status: "error", message: String(err) }), { status: 500, headers: { "Content-Type": "application/json" }, }); } });
